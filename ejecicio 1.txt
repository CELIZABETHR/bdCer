CREATE Table Tarea(
    id integer,
    nombre varchar,
    inicio integer,
    consumo decimal,
    fin integer,
    estado varchar,
    list_de_responsables integer,
    primary key (id));

CREATE table Usuario(
    Id integer,
    Nombre varchar,
    apellido varchar,
    telefono integer,
    primary key (Id));
    
CREATE TABLE  relacion(
     id_tarea integer,
     id_usuario integer);

CREATE TABLE Checklist(
    id integer,
    nombre varchar,
    estado varchar,
    primary key (Id));

CREATE TABLE Comentarios(
   id integer,
   Detalle varchar,
   responsable varchar,
   fecha date,
   primary key (Id));

CREATE TABLE Relacion2(
  id_checkist,
  id_comentarios);
